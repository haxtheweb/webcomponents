Basic hax-body-behaviors as used in lrn-math

Here is the way to do wiring with HAX in your element.

        import { HAXWiring } from "@haxtheweb/hax-body-behaviors/lib/HAXWiring.js";
        ...
        connectedCallback() {
          // Establish hax properties if they exist
          let props = {
            canScale: true,
            
            canEditSource: true,
            gizmo: {
              title: "Math",
              description: "Present math in a nice looking way.",
              icon: "places:all-inclusive",
              color: "grey",
              groups: ["Content"],
              handles: [
                {
                  type: "math",
                  math: "mathText"
                },
                {
                  type: "inline",
                  text: "mathText"
                }
              ],
              meta: {
                author: "HAXTheWeb core team"
              }
            },
            settings: {
              quick: [],
              configure: [
                {
                  slot: "",
                  title: "Math",
                  description: "Math",
                  inputMethod: "code-editor",
                  icon: "editor:format-quote"
                }
              ],
              advanced: []
            }
          };
          let wiring = new HAXWiring();
          wiring.setup(
            props,
            "lrn-math",
            this
          );
        }
